Drop Table if EXISTS history;
Drop Table if EXISTS products_orders;
Drop Table if EXISTS orders;
Drop Table if EXISTS products;
Drop Table if EXISTS buyer;
Drop Table if EXISTS seller;
Drop Table if EXISTS users;

CREATE TABLE Users (
id INT(6) AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(30) NOT NULL,
password VARCHAR(30) NOT NULL,
email VARCHAR(50),
address VARCHAR(30),
usericon VARCHAR(500),
status CHAR(7),

CHECK ((status='Buyer')OR (status='Seller')OR (status = 'Admin'))
);

CREATE TABLE Seller (
id INT(6) AUTO_INCREMENT PRIMARY KEY,
sales decimal(12,2),
name VARCHAR(30),
backgroundurl VARCHAR(500),
FOREIGN KEY (id) REFERENCES Users(id)
);

CREATE TABLE Buyer (
id INT(6) AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(30),
FOREIGN KEY (id) REFERENCES Users(id)
);

CREATE TABLE Products (
productid INT(6) AUTO_INCREMENT PRIMARY KEY,
product_status VARCHAR(30) DEFAULT "active",
name VARCHAR(30),
sellerid INT(6),
price decimal(7,2),
description VARCHAR(300),
category VARCHAR(15),
quantity INT(6),
img VARCHAR(500),
rating decimal(2,1),
FOREIGN KEY (sellerid) REFERENCES Seller(id)
);

CREATE TABLE Orders (
userid INT(6),
productid INT(6),
status VARCHAR(30),
quantity int,
orderid INT(6) auto_increment,
amount INT(6),
date DateTime,
PRIMARY KEY(orderid),
FOREIGN KEY (userid) REFERENCES Buyer(id)   
);

CREATE TABLE products_orders (
productid INT(6),
orderid INT(6),
PRIMARY KEY(productid,orderid),
FOREIGN KEY (productid) REFERENCES Products(productid),
FOREIGN KEY (orderid) REFERENCES Orders(orderid)
);

CREATE TABLE History (
sellerid INT(6),
buyerid INT(6),
productid INT(6),
orderid INT(6),
FOREIGN KEY (sellerid) REFERENCES Seller(id),
FOREIGN KEY (buyerid) REFERENCES Buyer(id),
FOREIGN KEY (productid,orderid) REFERENCES products_orders(productid,orderid)
);
